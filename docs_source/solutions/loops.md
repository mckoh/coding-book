```python

# create first loop with values from 1 to 10 (11 not included!)
for i in range(1, 11):

    # create first loop with values from 1 to 10 (11 not included!)
    for j in range(1, 11):

        # calculate the product of i and j and store for latter use
        product = i * j

        # print the inputs and the result to console
        print(f"{i} * {j} = {product}")
```
